def port_channel():
	cli.configurep(['interface Port-channel1',
	'ip add 10.10.10.2 255.255.255.252')]

def config_switchport():
	cli.configurep(['interface Ge0/0',
	'no switchport',
	'duplex full,
	'channel-group 1 mode on')]
	cli.configurep(['interface Ge0/1',
	'no switchport',
	'duplex full',
	'no ip add',
	'channel-group 1 mode on')]
	cli.configurep(['interface Ge0/2',
	'no switchport',
	'duplex full',
	'ip add 172.16.6.1 255.255.255.0',
	'duplex full')]
	cli.configurep(['interface Ge0/3',
	'no switchport',
	'duplex full',
	'ip add 172.16.7.1 255.255.255.0',
	'duplex full')]
	cli.configurep(['interface Ge1/0',
	'no switchport',
	'duplex full',
	'ip add 172.16.8.1 255.255.255.0',
	'duplex full')]

def config_routing():
	cli.configurep(['router eigrp 1',
	'network 10.10.10.0 0.0.0.255',

	'network 172.16.7.0 0.0.0.255',
	'network 172.16.8.0 0.0.0.255')]
